## Installing PoloCloud

Choose your preferred installation method:

<Tabs items={['Local', 'Docker']}>
  <Tab value="Local">
    <div className="text-muted-foreground mb-4">
      Install PoloCloud directly on your local machine. Requires Java 17 or higher.
    </div>
    
    <div className="mb-4">
      <a 
        href="https://github.com/PoloCloud/PoloCloud/releases" 
        target="_blank"

      >
        <Download className="w-3.5 h-3.5" />
        Download PoloCloud
      </a>
    </div>
    
    <Tabs items={['Simple', 'Production']} defaultValue="Production">
      <Tab value="Simple">
        <div className="relative group">
          <DynamicCodeBlock 
            lang="bash" 
            code='java -jar polocloud-launcher.jar'
            className="rounded-lg border border-border/50 bg-muted/30"
          />
        </div>
      </Tab>
      <Tab value="Production">
        <div className="relative group">
          <DynamicCodeBlock 
            lang="bash" 
            code='chmod +x ./start.sh
./start.sh'
            className="rounded-lg border border-border/50 bg-muted/30"
          />
        </div>
      </Tab>
    </Tabs>
    
    <div className="text-muted-foreground mt-4">
      This command starts PoloCloud with the specified configuration mode.
    </div>
  </Tab>
  
  <Tab value="Docker">
    <div className="text-muted-foreground mb-4">
      Run PoloCloud in a Docker container. Requires Docker and Docker Compose.
    </div>
    
    <div className="text-muted-foreground mb-3">Pull the latest PoloCloud Docker image:</div>
    <div className="relative group">
      <DynamicCodeBlock 
        lang="bash" 
        code='docker pull ghcr.io/httpmarco/polocloud:latest'
        className="rounded-lg border border-border/50 bg-muted/30"
      />
    </div>
    
    <div className="text-muted-foreground mb-3 mt-4">Run PoloCloud in a Docker container:</div>
    <div className="relative group">
      <DynamicCodeBlock 
        lang="bash" 
        code='docker run -d --name polocloud -p 8080:8080 ghcr.io/httpmarco/polocloud:latest'
        className="rounded-lg border border-border/50 bg-muted/30"
      />
    </div>
  </Tab>
</Tabs>
